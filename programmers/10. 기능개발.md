```
def solution(progresses, speeds):
    answer = []
    days = []
    
    for i in range(len(progresses)):
        remaining = 100 - progresses[i]
        days.append(remaining // speeds[i] + (1 if remaining % speeds[i] != 0 else 0))
    
    release_day = days[0]
    count = 0
    
    for day in days:
        if day <= release_day:
            count += 1
        else:
            answer.append(count)
            count = 1
            release_day = day
            
    answer.append(count)
    return answer
```
